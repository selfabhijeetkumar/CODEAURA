## 🗃️ Database Schema

<br/>

```sql
-- ═══════════════════════════════════════════════════════════════════
--                     CODEAURA DATABASE SCHEMA
--                   Supabase PostgreSQL — Full Blueprint
-- ═══════════════════════════════════════════════════════════════════


-- ─────────────────────────────────────────────────────────────────
-- TABLE 01: sessions
-- Stores every code analysis session a user runs.
-- The core data entity of CODEAURA.
-- ─────────────────────────────────────────────────────────────────

CREATE TABLE sessions (
  id                uuid          PRIMARY KEY DEFAULT gen_random_uuid(),
  session_name      text          NOT NULL,
  code              text          NOT NULL,
  language          text,
  steps             jsonb,        -- Full execution step array from Gemini
  quality_score     integer,      -- 0–100 AI quality rating
  quiz_score        integer,      -- User's quiz performance
  bug_count         integer,      -- Total bugs detected
  complexity_time   text,         -- e.g. "O(n log n)"
  complexity_space  text,         -- e.g. "O(n)"
  share_token       uuid          DEFAULT gen_random_uuid(),  -- For sharing
  created_at        timestamptz   DEFAULT now(),
  updated_at        timestamptz   DEFAULT now()
);

-- Index for fast share link lookups
CREATE INDEX idx_sessions_share_token ON sessions(share_token);


-- ─────────────────────────────────────────────────────────────────
-- TABLE 02: session_notes
-- User annotations on individual execution steps.
-- Powers the "learn by writing" feature.
-- ─────────────────────────────────────────────────────────────────

CREATE TABLE session_notes (
  id                uuid          PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id        uuid          REFERENCES sessions(id) ON DELETE CASCADE,
  step_number       integer       NOT NULL,
  ai_explanation    text,         -- AI's deep-dive explanation of this step
  user_note         text,         -- User's personal annotation
  created_at        timestamptz   DEFAULT now()
);

-- Index for fast session note lookups
CREATE INDEX idx_notes_session_id ON session_notes(session_id);


-- ─────────────────────────────────────────────────────────────────
-- TABLE 03: leaderboard_stats
-- Global competition rankings based on code quality scores.
-- ─────────────────────────────────────────────────────────────────

CREATE TABLE leaderboard_stats (
  id                uuid          PRIMARY KEY DEFAULT gen_random_uuid(),
  username          text          NOT NULL UNIQUE,
  total_sessions    integer       DEFAULT 0,
  average_score     float         DEFAULT 0.0,
  best_score        integer       DEFAULT 0,
  total_bugs_fixed  integer       DEFAULT 0,
  rank_badge        text          DEFAULT 'Getting Started',
  -- Badge levels:
  -- 'Getting Started'  → avg 0–40
  -- 'Rising Coder'     → avg 41–60
  -- 'Sharp Developer'  → avg 61–75
  -- 'Elite Programmer' → avg 76–90
  -- 'Code Overlord'    → avg 91–100
  last_active       timestamptz   DEFAULT now(),
  created_at        timestamptz   DEFAULT now()
);

-- Index for leaderboard ranking queries
CREATE INDEX idx_leaderboard_avg_score ON leaderboard_stats(average_score DESC);
CREATE INDEX idx_leaderboard_best_score ON leaderboard_stats(best_score DESC);
